select rlt.RLTSHP_ID AS RLTSHP_ID,
sub_acc.VNDR_ACCT_ID AS VNDR_ACCT_ID,
sub_acc.ACCT_ID AS ACCT_ID,
sub_acc.BR_NO as BR_NO,
count(distinct sub_acc_type.ACCT_TY_CD) as subAccountTypeCount
from PROD.RLTSHP_13120 rlt
join PROD.ACCT_FUNC_GRP_DTL_13088 grp on rlt.ACCT_FUNC_GRP_ID = grp.ACCT_FUNC_GRP_ID
join PROD.SUBACCT_SVC_12569 as sub_acct_svc on grp.ACCT_ID = sub_acct_svc.ACCT_ID
join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID = sub_acct_svc.ACCT_ID
join PROD.SUBACCT_TYPE_12528 as sub_acc_type on sub_acc.ACCT_ID = sub_acc_type.ACCT_ID
join PROD.ACCOUNT_12280 as acct on acct.ACCT_ID = sub_acct_svc.ACCT_ID
join PROD.ACCT_OWN_RUL_12520 acc_own_rule on acc_own_rule.ACCT_OWN_RULE_ID =  acct.ACCT_OWN_RULE_ID
join PROD.OWN_CLASS_12680 cls on  acc_own_rule.ACCT_OWN_RULE_ID = cls.ACCT_OWN_RULE_ID
where rlt.ISO_CTRY_CD = 'US'
and cls.ACCT_CLASS_CD ='01'
and sub_acct_svc.ACTV_IND = 'Y'
and sub_acc.AUTH_FORM_IND = 'Y'
and sub_acc.BR_NO = 4454
and sub_acc.ACCT_CLS_UT is NULL
and sub_acct_svc.ACCT_SVC_TY_CD='SE'
and sub_acc_type.ACCT_TY_CD in (1,2,4) --cash, margin, loan
and rlt.RLTSHP_ID <> 44899208
and sub_acc.acct_id in 
(               select a.acct_id
                from
                    PROD.ACCOUNT_12280 a
                    left outer join PROD.acct_rst_10214 v on v.account_no= a.cust_no
                    inner join PROD.ACCT_OWN_RUL_12520 acc_own_rule on acc_own_rule.ACCT_OWN_RULE_ID =  a.ACCT_OWN_RULE_ID
                    inner join PROD.OWN_CLASS_12680 cls on  acc_own_rule.ACCT_OWN_RULE_ID =  cls.ACCT_OWN_RULE_ID
                    inner join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=a.ACCT_ID
                where
                    v.rst_rsn_cd is not null
                    and v.rst_rsn_cd in ( '72T', 'CIPR', 'AUTH', 'WARN', 'CO', 'GARN', 'LEVY', 'FOR2', 'RISK', 'MAIL', 'FTRN' )
                    and cls.ACCT_CLASS_CD ='01'
                    and sub_acc.BR_NO = 4454
                    and a.ACCT_CLS_DA is null
)
group by rlt.RLTSHP_ID, sub_acc.VNDR_ACCT_ID, sub_acc.ACCT_ID, sub_acc.BR_NO
having count(distinct sub_acc_type.ACCT_TY_CD) = 3
fetch first 10 rows only;
