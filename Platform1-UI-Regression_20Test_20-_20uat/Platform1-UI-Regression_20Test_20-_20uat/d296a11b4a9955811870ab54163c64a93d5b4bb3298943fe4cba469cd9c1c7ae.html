acct.ACCT_ID, acct.CUST_NO,

where
acct.TAX_RPT_CTRY_CD='US'
and acct.ACCT_OWN_RULE_ID=312
and sub_acc.br_no   = 9979

--account is not deleted and closed
and acct.LGCL_DEL_DA is null
and acct.ACCT_CLS_DA is null
--owner is not deceased
and contact.DEATH_DA is  null
--inherited = Y
and acct.dcdnt_retr_ind = 'N'

--and hq_system_restrictions.ACH_CR_ALLW_IND='N'

    and acct.AUTH_FORM_IND = 'Y' and acct_svc.ACCT_SVC_TY_CD ='SE'
 and acct_svc.ACCT_SVC_TY_CD not in ('TL')


--account is [not] restricted by restrictions from list
and not acct.cust_no in (
    select distinct acct.CUST_NO

    from PROD.ACCOUNT_12280 as acct
    join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=acct.ACCT_ID

    acct.TAX_RPT_CTRY_CD='US'
