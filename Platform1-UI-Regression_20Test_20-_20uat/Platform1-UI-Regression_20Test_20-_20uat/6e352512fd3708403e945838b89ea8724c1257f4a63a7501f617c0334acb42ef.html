from PROD.RLTSHP_13120 rlt
join PROD.ACCT_FUNC_GRP_DTL_13088 grp on rlt.ACCT_FUNC_GRP_ID = grp.ACCT_FUNC_GRP_ID
join PROD.ACCOUNT_12280 as acct on grp.ACCT_ID = acct.ACCT_ID
join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=acct.ACCT_ID
join PROD.SUBACCT_SVC_12569 as acct_svc on acct_svc.ACCT_ID=acct.ACCT_ID

join PROD.US_SUB_ACCT_13603 as us_sub_acc on sub_acc.ACCT_ID = us_sub_acc.ACCT_ID
left outer join PROD.DB_CR_ALLW_11903 as hq_system_restrictions on hq_system_restrictions.ACCT_ID = acct.ACCT_ID

where
acct.TAX_RPT_CTRY_CD='US'
--and acct.ACCT_OWN_RULE_ID=
and sub_acc.br_no   = 9979

--owner is not deceased
and contact.DEATH_DA is  null

and acct.dcdnt_retr_ind = 'N'

--and (hq_system_restrictions.ACH_CR_ALLW_IND is null or hq_system_restrictions.ACH_CR_ALLW_IND='Y')
and hq_system_restrictions.ACH_CR_ALLW_IND='N'
