join PROD.CONTACT_12250 as contact on contact.CON_ID  = sub_acc.TAX_RPT_CON_ID

where rlt.ISO_CTRY_CD = 'US'
and cls.ACCT_CLASS_CD ='01'
 and sub_acct_svc.ACTV_IND = 'Y'

and sub_acc.BR_NO = 9979
and sub_acc.ACCT_CLS_UT is NULL

select
rlt.RLTSHP_ID

from PROD.SUB_ACCOUNT_12512 as sub_acc
join PROD.RLTSHP_13120 as rlt on rlt.ACCT_FUNC_GRP_ID = acctgrp.ACCT_FUNC_GRP_ID
join PROD.ACCT_FUNC_GRP_DTL_13088 as acctgrp on acctgrp.ACCT_ID=sub_acc.ACCT_ID

where sub_acc.BR_NO = 9979
and sub_acc.ACCT_CLS_UT is null

group by rlt.RLTSHP_ID

)

                    v.rst_rsn_cd is not null
