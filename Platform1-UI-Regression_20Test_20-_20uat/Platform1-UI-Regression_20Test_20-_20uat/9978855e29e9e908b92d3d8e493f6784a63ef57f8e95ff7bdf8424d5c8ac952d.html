select rlt.RLTSHP_ID AS RLTSHP_ID,
sub_acc.VNDR_ACCT_ID AS VNDR_ACCT_ID,
sub_acc.ACCT_ID AS ACCT_ID,

count(distinct sub_acc_type.ACCT_TY_CD) as subAccountTypeCount,
sum(sub_acc_type.ACCT_TY_CD) as subAccountsSum

from PROD.RLTSHP_13120 rlt
join PROD.ACCT_FUNC_GRP_DTL_13088 grp on rlt.ACCT_FUNC_GRP_ID = grp.ACCT_FUNC_GRP_ID

join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID = sub_acct_svc.ACCT_ID

join PROD.CONTACT_12250 as contact on contact.CON_ID  = sub_acc.TAX_RPT_CON_ID

where rlt.ISO_CTRY_CD = 'US'
and cls.ACCT_CLASS_CD ='01'
 and sub_acct_svc.ACTV_IND = 'Y'
 and sub_acc.AUTH_FORM_IND = 'Y'
and sub_acc.BR_NO = 4454
and sub_acc.ACCT_CLS_UT is NULL
and sub_acc.acct_rst_ind = 'N'
 and sub_acct_svc.ACCT_SVC_TY_CD='SE'
--owner is not deceased
and contact.DEATH_DA is null
--inherited = N
 and acct.dcdnt_retr_ind = 'N'
-- and acct.ACCT_ACTV_IND='N'

and sub_acc.acct_id not in
(               select a.acct_id
                from
                    PROD.ACCOUNT_12280 a
                    left outer join PROD.acct_rst_10214 v on v.account_no= a.cust_no
                    inner join PROD.ACCT_OWN_RUL_12520 acc_own_rule on acc_own_rule.ACCT_OWN_RULE_ID =  a.ACCT_OWN_RULE_ID

                    inner join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=a.ACCT_ID
                where
                    v.rst_rsn_cd is not null

                    and a.ACCT_CLS_DA is null
)
