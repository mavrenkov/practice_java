join PROD.ACCOUNT_12280 as acc on acc.ACCT_ID = t.ACCT_ID
where rlt.ALL_ACCT_CLS_IND = 'N'
AND acc.ACCT_CLS_DA IS NULL
AND acc.LGCL_DEL_DA IS NULL
FETCH FIRST 5 ROWS ONLY
