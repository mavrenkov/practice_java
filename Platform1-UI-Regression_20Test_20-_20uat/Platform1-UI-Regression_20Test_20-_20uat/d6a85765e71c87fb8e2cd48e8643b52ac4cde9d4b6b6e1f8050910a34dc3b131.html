select
acct.ACCT_ID, acct.CUST_NO,
rlt.RLTSHP_ID AS RLTSHP_ID,
acct_svc.ACCT_SVC_TY_CD, acct_svc.ACCT_SVC_STAT_CD, acct.ACCT_OWN_RULE_ID,
contact.CON_ID, contact.DEATH_DA,
hq_system_restrictions.ACH_DB_ALLW_IND, hq_system_restrictions.ACH_CR_ALLW_IND,
sub_acc.BR_NO,
us_sub_acc.EDJ_ACCT_PRFD_NA as ACCT_PREFERRED_NAME

from PROD.RLTSHP_13120 rlt
join PROD.ACCT_FUNC_GRP_DTL_13088 grp on rlt.ACCT_FUNC_GRP_ID = grp.ACCT_FUNC_GRP_ID
join PROD.ACCOUNT_12280 as acct on grp.ACCT_ID = acct.ACCT_ID
join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=acct.ACCT_ID
join PROD.SUBACCT_SVC_12569 as acct_svc on acct_svc.ACCT_ID=acct.ACCT_ID
join PROD.CONTACT_12250 as contact on contact.CON_ID  = sub_acc.TAX_RPT_CON_ID
join PROD.US_SUB_ACCT_13603 as us_sub_acc on sub_acc.ACCT_ID = us_sub_acc.ACCT_ID
left outer join PROD.DB_CR_ALLW_11903 as hq_system_restrictions on hq_system_restrictions.ACCT_ID = acct.ACCT_ID

where
acct.TAX_RPT_CTRY_CD='US'
and acct.ACCT_OWN_RULE_ID=309
and sub_acc.br_no   = 9979

--account is not deleted and closed
and acct.LGCL_DEL_DA is null
and acct.ACCT_CLS_DA is null
--owner is not deceased
and contact.DEATH_DA is  null
--inherited = Y
and acct.dcdnt_retr_ind = 'N'

---- Y[hqEjToBank=true] = client is allowed to initiate transfers from this Edward Jones account to their bank account
and (hq_system_restrictions.ACH_DB_ALLW_IND is null  or hq_system_restrictions.ACH_DB_ALLW_IND='Y')
--and hq_system_restrictions.ACH_DB_ALLW_IND='N'
---- Y[hqEjToBank=true] = client is allowed to initiate transfers from their bank account to this Edward Jones account
and (hq_system_restrictions.ACH_CR_ALLW_IND is null or hq_system_restrictions.ACH_CR_ALLW_IND='Y')
--and hq_system_restrictions.ACH_CR_ALLW_IND='N'

    and acct.AUTH_FORM_IND = 'Y' and acct_svc.ACCT_SVC_TY_CD ='TL'

--account is [not] restricted by restrictions from list
and not acct.cust_no in (
    select distinct acct.CUST_NO

    from PROD.ACCOUNT_12280 as acct
    join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=acct.ACCT_ID
    left outer join PROD.acct_rst_10214 v on v.account_no=acct.cust_no

    where
    acct.TAX_RPT_CTRY_CD='US'

    and acct.ACCT_OWN_RULE_ID=309
    and sub_acc.br_no   = 9979
    and acct.LGCL_DEL_DA is null
    and acct.ACCT_CLS_DA is null
    and v.rst_rsn_cd in ('CIPR','COPD','FTRN','MAIL','RISK','WARN','72T','SSN','AUTH','RESTRICTION')
    and v.RST_START_DA<'2025-10-06'
)
