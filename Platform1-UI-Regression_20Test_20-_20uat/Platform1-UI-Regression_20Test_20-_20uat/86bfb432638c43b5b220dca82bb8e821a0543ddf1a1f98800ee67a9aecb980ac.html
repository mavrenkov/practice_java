select
accountsData.RLTSHP_ID AS RELID, accountsData.BR_NO,
count(distinct accountsData.TAX_RPT_TIN_NO) as SSN_COUNT,
count(distinct accountsData.acct_ID) as ACCOUNTS_COUNT,
count(case (accountsData.COUNT_CASH + accountsData.COUNT_MARGIN*2 + accountsData.COUNT_LOAN*4) when 7 then 1 end) as CASH_MARGIN_LOAN_ACC_COUNT,
min(accountsData.TAX_RPT_TIN_NO) as ssn1, max(accountsData.TAX_RPT_TIN_NO) as ssn2,
min(accountsData.acct_ID) as acct1, max(accountsData.acct_ID) as acct2,
max(accountsData.COUNT_CASH) as cash,
max(accountsData.COUNT_MARGIN) as margin,
max(accountsData.COUNT_LOAN) as loan,
min(case (accountsData.COUNT_CASH + accountsData.COUNT_MARGIN*2 + accountsData.COUNT_LOAN*4) when 1 then accountsData.acct_ID end) as account1_cash_only,
max(case (accountsData.COUNT_CASH + accountsData.COUNT_MARGIN*2 + accountsData.COUNT_LOAN*4) when 1 then accountsData.acct_ID end) as account2_cash_only,
min(case (accountsData.COUNT_CASH + accountsData.COUNT_MARGIN*2 + accountsData.COUNT_LOAN*4) when 7 then accountsData.acct_ID end) as account1_cash_margin_loan,
max(case (accountsData.COUNT_CASH + accountsData.COUNT_MARGIN*2 + accountsData.COUNT_LOAN*4) when 7 then accountsData.acct_ID end) as account2_cash_margin_loan

from (
        select
        rlt.RLTSHP_ID AS RLTSHP_ID, sub_acc.BR_NO as BR_NO,
        sub_acc.TAX_RPT_TIN_NO as TAX_RPT_TIN_NO,
        sub_acc_svc.acct_ID as ACCT_ID,
        account.acct_own_rule_id as acct_own_rule_id,
        count(distinct sub_acc_type1.ACCT_ID) as COUNT_CASH,
        count(distinct sub_acc_type2.ACCT_ID) as COUNT_MARGIN,
        count(distinct sub_acc_type4.ACCT_ID) as COUNT_LOAN

        from PROD.SUB_ACCOUNT_12512 as sub_acc
        join PROD.SUBACCT_SVC_12569 sub_acc_svc on sub_acc.ACCT_ID = sub_acc_svc.ACCT_ID
        join PROD.RLTSHP_13120 as rlt on rlt.ACCT_FUNC_GRP_ID = acctgrp.ACCT_FUNC_GRP_ID
        join PROD.ACCT_FUNC_GRP_DTL_13088 as acctgrp on acctgrp.ACCT_ID=sub_acc.ACCT_ID
        join PROD.ACCOUNT_12280 account on account.ACCT_ID=sub_acc.ACCT_ID
        left outer join PROD.SUBACCT_TYPE_12528 as sub_acc_type1 on (sub_acc.ACCT_ID = sub_acc_type1.ACCT_ID and sub_acc_type1.ACCT_TY_CD = 1) --cash
        left outer join PROD.SUBACCT_TYPE_12528 as sub_acc_type2 on (sub_acc.ACCT_ID = sub_acc_type2.ACCT_ID and sub_acc_type2.ACCT_TY_CD = 2) --margin
        left outer join PROD.SUBACCT_TYPE_12528 as sub_acc_type4 on (sub_acc.ACCT_ID = sub_acc_type4.ACCT_ID and sub_acc_type4.ACCT_TY_CD = 4) --loan

        where
        -- sub_acc.ACCT_OWN_SUBTY_CD is not NULL and
        sub_acc.ACCT_CLS_UT is NULL
        and sub_acc.BR_NO in (4454)
        and sub_acc_svc.ISO_CTRY_CD='US'
        and account.acct_own_rule_id in (309)
        and account.dcdnt_retr_ind = 'N'
        and rlt.RLTSHP_ID not in (82130780)
         and sub_acc_svc.ACCT_SVC_TY_CD in ('SE')
         and sub_acc_svc.ACCT_SVC_STAT_CD in ('A', 'C')

        and sub_acc.acct_id not in

    (               select a.acct_id
                    from
                        PROD.ACCOUNT_12280 a
                        left outer join PROD.acct_rst_10214 v on v.account_no= a.cust_no
                        inner join PROD.ACCT_OWN_RUL_12520 acc_own_rule on acc_own_rule.ACCT_OWN_RULE_ID =  a.ACCT_OWN_RULE_ID
                        inner join PROD.OWN_CLASS_12680 cls on  acc_own_rule.ACCT_OWN_RULE_ID =  cls.ACCT_OWN_RULE_ID
                        inner join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=a.ACCT_ID
                    where
                        v.rst_rsn_cd is not null
                        and v.rst_rsn_cd in ('AUTH', 'DOCU', 'WARN', 'CIPR', 'COPD', 'CO', 'RISK', 'SSN', 'FTRN', '72T', 'MISC', 'MAIL', 'FOR', 'FOR2', 'R471', 'R407','DEATH')
                        and a.acct_own_rule_id in (309)
                        and sub_acc.BR_NO in (4454)
                        and a.ACCT_CLS_DA is null
                        and v.RST_START_DA<'2025-10-06'
    )

        group by
            rlt.RLTSHP_ID, sub_acc.BR_NO,
            sub_acc.TAX_RPT_TIN_NO,
            sub_acc_svc.acct_ID,
            account.acct_own_rule_id

        order by sub_acc.BR_NO, rlt.RLTSHP_ID

        fetch first 3000 rows only
) as accountsData
group by accountsData.RLTSHP_ID, accountsData.BR_NO
having
count(distinct accountsData.ACCT_ID)>=2
and count(distinct accountsData.TAX_RPT_TIN_NO) > 1
and count(distinct accountsData.acct_own_rule_id) = 1
and max(accountsData.COUNT_CASH)=1
and max(accountsData.COUNT_MARGIN)=1
and max(accountsData.COUNT_LOAN)=1
