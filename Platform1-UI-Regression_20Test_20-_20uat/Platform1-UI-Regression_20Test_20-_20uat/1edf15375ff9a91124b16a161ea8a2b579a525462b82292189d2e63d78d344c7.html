select
rlt.RLTSHP_ID AS RELID, sub_acc.BR_NO,
count(distinct sub_acc.TAX_RPT_TIN_NO) as SSN_COUNT,

min(sub_acc.TAX_RPT_TIN_NO) as ssn1, max(sub_acc.TAX_RPT_TIN_NO) as ssn2,
min(sub_acc_svc.acct_ID) as acct1, max(sub_acc_svc.acct_ID) as acct2

from PROD.SUB_ACCOUNT_12512 as sub_acc
join PROD.SUBACCT_SVC_12569 sub_acc_svc on sub_acc.ACCT_ID = sub_acc_svc.ACCT_ID
join PROD.RLTSHP_13120 as rlt on rlt.ACCT_FUNC_GRP_ID = acctgrp.ACCT_FUNC_GRP_ID
join PROD.ACCT_FUNC_GRP_DTL_13088 as acctgrp on acctgrp.ACCT_ID=sub_acc.ACCT_ID
join PROD.ACCOUNT_12280 account on account.ACCT_ID=sub_acc.ACCT_ID
where
-- sub_acc.ACCT_OWN_SUBTY_CD is not NULL and
sub_acc.ACCT_CLS_UT is NULL
and sub_acc.BR_NO in (1626, 2153, 4369, 9005, 3434, 94674, 87782, 7822, 5897, 3997, 19864, 99785, 99732, 99464, 99447, 98594, 95528, 4454, 9979)
and sub_acc_svc.ISO_CTRY_CD='US'
and account.acct_own_rule_id in (309)
and account.dcdnt_retr_ind = 'N'
 and sub_acc_svc.ACCT_SVC_TY_CD in ('SE')
 and sub_acc_svc.ACCT_SVC_STAT_CD in ('A', 'C')
-- and rlt.RLTSHP_ID in ()

and sub_acc.acct_id not in 
(               select a.acct_id
                from
                    PROD.ACCOUNT_12280 a
                    left outer join PROD.acct_rst_10214 v on v.account_no= a.cust_no
                    inner join PROD.ACCT_OWN_RUL_12520 acc_own_rule on acc_own_rule.ACCT_OWN_RULE_ID =  a.ACCT_OWN_RULE_ID
                    inner join PROD.OWN_CLASS_12680 cls on  acc_own_rule.ACCT_OWN_RULE_ID =  cls.ACCT_OWN_RULE_ID
                    inner join PROD.SUB_ACCOUNT_12512 as sub_acc on sub_acc.ACCT_ID=a.ACCT_ID
                where
                    v.rst_rsn_cd is not null
                    and v.rst_rsn_cd in ('AUTH', 'DOCU', 'WARN', 'CIPR', 'COPD', 'CO', 'RISK', 'SSN', 'FTRN', '72T', 'MISC', 'MAIL', 'FOR', 'FOR2', 'R471', 'R407','DEATH')
                                   and a.acct_own_rule_id in (309)
                    and sub_acc.BR_NO in (1626, 2153, 4369, 9005, 3434, 94674, 87782, 7822, 5897, 3997, 19864, 99785, 99732, 99464, 99447, 98594, 95528, 4454, 9979)
                    and a.ACCT_CLS_DA is null
                    and v.RST_START_DA<'2025-10-06'
)

group by rlt.RLTSHP_ID, sub_acc.BR_NO
having 
count(distinct sub_acc_svc.ACCT_ID)>1
